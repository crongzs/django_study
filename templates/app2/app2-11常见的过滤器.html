{{ v1|add:v2 }}

cut 过滤器

{{ v3|cut:' ...' }}

date 过滤器

{{ now|date:'y M d D g a' }}

{{ v4|default:'betelnut' }}

{{ v6|first }}

{{ v6|last }}

{{ v9|safe }}
